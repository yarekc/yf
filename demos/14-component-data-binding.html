How it works:

<!-- 2. Bind inputs to data -->

<input type="text" data-bind="userTitle">

<!-- 3. Use component with data-bound props -->
<profile-card 
  data-name="userName"
  data-title="userTitle"
  data-email="userEmail"
  data-location="userLocation">
</profile-card>

<script>
  // Initialize data
  setData('userName', 'John Doe');

  setData('userEmail', 'john@example.com');

</script>
