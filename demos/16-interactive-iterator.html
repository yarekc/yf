How it works:

<!-- Input for new task -->
<input type="text" data-bind="newTask">
<button onclick="addTask()">Add</button>

<!-- Show count -->
<div>Total: {tasks.length}</div>

<!-- Show when empty -->
<div show="{tasks.length == 0}">No tasks yet!</div>

<!-- Loop through tasks -->
<div data-for="tasks as task">
  <div>
    {task.text}
    <button onclick="deleteTask({task.id})">Delete</button>
  </div>
</div>

<script>
  let nextId = 1;
  setData('tasks', []);
  
  function addTask() {
    const text = getData('newTask');
    if (!text.trim()) return;
    
    const tasks = getData('tasks');
    setData('tasks', [
      ...tasks,
      { id: nextId++, text: text }
    ]);
    setData('newTask', '');
  }
  
  function deleteTask(id) {
    const tasks = getData('tasks');
    setData('tasks', tasks.filter(t => t.id !== id));
  }
</script>
