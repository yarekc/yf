How it works:

<!-- Navigation buttons -->
<button onclick="setState('app', 'home')">Home</button>
<button onclick="setState('app', 'about')">About</button>
<button onclick="setState('app', 'contact')">Contact</button>

<!-- Container with ID -->
<div id="app">
  <!-- Each page has a state attribute -->
  <div state="home">
    <h2>Home Page</h2>
    <p>Home content...</p>
  </div>

  <div state="about">
    <h2>About Page</h2>
    <p>About content...</p>
  </div>

  <div state="contact">
    <h2>Contact Page</h2>
    <p>Contact content...</p>
  </div>
</div>

<script>
  // Set initial state
  setState('app', 'home');
</script>
