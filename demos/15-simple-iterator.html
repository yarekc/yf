How it works:

<!-- Display array length -->
<div>Total: {products.length}</div>

<!-- Loop through array with data-for -->
<div data-for="products as product">
  <div class="item-card">
    <strong>{product.name}</strong>
    <p>Category: {product.category}</p>
    <p>${product.price}</p>
  </div>
</div>

<script>
  // Define array of products
  setData('products', [
    { name: 'Laptop', category: 'Electronics', price: 999 },
    { name: 'Mouse', category: 'Electronics', price: 29 },
    { name: 'Desk Chair', category: 'Furniture', price: 199 }
  ]);
</script>
